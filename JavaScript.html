O que é JavaScript?

JavaScript é uma linguagem de programação de alto nível, interpretada e multiparadigma, utilizada principalmente para adicionar interatividade às páginas web.

Características principais:

  • Linguagem de script: Executada no lado do cliente (navegador)
  • Multiparadigma: Suporta programação funcional, orientada a objetos e imperativa
  • Ecossistema rico: NPM, frameworks e bibliotecas poderosas
  • Universal: Funciona em todos os navegadores modernos e no servidor (Node.js)

O JavaScript trabalha em conjunto com HTML (estrutura) e CSS (estilo), formando a tríade fundamental do desenvolvimento web front-end.

Sintaxe Básica

JavaScript possui uma sintaxe similar a outras linguagens como C, Java e C#.

Variáveis e Tipos


// Declaração de variáveis (ES6+)
let nome = 'João'; // String
const idade = 30;  // Number (constante)
var antigo = true; // Boolean (modo antigo)

// Tipos de dados
let texto = "Texto";       // String
let numero = 42;           // Number
let booleano = false;      // Boolean
let nulo = null;           // Null
let indefinido;            // Undefined
let objeto = {             // Object
  chave: 'valor'
};
let array = [1, 'dois'];   // Array
let funcao = function() {}; // Function

Operadores


// Aritméticos
let soma = 10 + 5;    // 15
let subtracao = 10 - 5; // 5
let multiplicacao = 10 * 5; // 50
let divisao = 10 / 5;  // 2
let modulo = 10 % 3;   // 1 (resto da divisão)

// Comparação
let igual = 5 == '5';  // true (valor igual)
let estritamenteIgual = 5 === '5'; // false (valor e tipo)
let diferente = 5 != '5'; // false
let maiorQue = 10 > 5; // true

// Lógicos
let e = true && false; // false
let ou = true || false; // true
let nao = !true;       // false

Estruturas de Controle

Condicionais


// if-else
if (idade >= 18) {
  console.log('Maior de idade');
} else {
  console.log('Menor de idade');
}

// switch-case
switch(diaDaSemana) {
  case 0:
    console.log('Domingo');
    break;
  case 1:
    console.log('Segunda');
    break;
  default:
    console.log('Outro dia');
}

Loops


// for loop
for (let i = 0; i < 5; i++) {
  console.log(i);
}

// while
let i = 0;
while (i < 5) {
  console.log(i);
  i++;
}

// for...of (arrays)
for (let item of array) {
  console.log(item);
}

// for...in (objetos)
for (let chave in objeto) {
  console.log(chave, objeto[chave]);
}

Funções

Funções são blocos de código reutilizáveis que realizam uma tarefa específica.

Declaração de Funções


// Declaração de função
function saudacao(nome) {
  return `Olá, ${nome}!`;
}

// Expressão de função
const soma = function(a, b) {
  return a + b;
};

// Arrow function (ES6+)
const quadrado = x => x * x;

// Parâmetros padrão
function multiplicar(a, b = 1) {
  return a * b;
}

Recursos Avançados


// Funções de alta ordem
function operacao(a, b, funcao) {
  return funcao(a, b);
}

operacao(5, 3, (x, y) => x + y); // 8

// Closure
function contador() {
  let count = 0;
  return function() {
    return ++count;
  };
}

const incrementar = contador();
incrementar(); // 1
incrementar(); // 2

Escopo e Hoisting


// Escopo global
let global = 'acessível em qualquer lugar';

function testeEscopo() {
  // Escopo de função
  let funcao = 'acessível apenas aqui';
  
  if (true) {
    // Escopo de bloco (let/const)
    let bloco = 'apenas neste bloco';
    var antigo = 'vaza para a função'; // var não tem escopo de bloco
  }
}

// Hoisting - declarações são movidas para o topo
console.log(x); // undefined (var)
var x = 5;
console.log(y); // ReferenceError (let/const)
let y = 10;

Objetos e Arrays

Objetos e arrays são estruturas fundamentais para organizar dados em JavaScript.

Objetos


// Criação de objeto
const pessoa = {
  nome: 'Maria',
  idade: 25,
  saudar: function() {
    console.log(`Olá, meu nome é ${this.nome}`);
  }
};

// Acesso a propriedades
console.log(pessoa.nome); // Maria
console.log(pessoa['idade']); // 25
pessoa.saudar(); // Olá, meu nome é Maria

// Adicionar/remover propriedades
pessoa.profissao = 'Desenvolvedora';
delete pessoa.idade;

// Object destructuring (ES6+)
const { nome, saudar } = pessoa;
saudar(); // Olá, meu nome é Maria

Arrays


// Criação de array
const frutas = ['maçã', 'banana', 'laranja'];

// Acesso
console.log(frutas[1]); // banana
console.log(frutas.length); // 3

// Métodos comuns
frutas.push('uva'); // adiciona no final
frutas.pop(); // remove do final
frutas.unshift('morango'); // adiciona no início
frutas.shift(); // remove do início

// Array destructuring (ES6+)
const [primeira, segunda] = frutas;
console.log(primeira); // maçã

// Métodos funcionais
frutas.forEach(fruta => console.log(fruta));
const maiusculas = frutas.map(f => f.toUpperCase());
const filtradas = frutas.filter(f => f.length > 5);

Manipulação do DOM

O DOM (Document Object Model) é a representação em objetos da estrutura HTML da página.

Seleção de Elementos


// Métodos tradicionais
document.getElementById('meuId');

document.getElementsByTagName('div');

// Métodos modernos (querySelector)
document.querySelector('#meuId'); // seletor CSS
document.querySelectorAll('.minhaClasse');

// Exemplos práticos
const botao = document.querySelector('button');
const paragrafos = document.querySelectorAll('p');
const formulario = document.forms['meuForm'];

Manipulação de Elementos


// Alterar conteúdo
elemento.textContent = 'Novo texto';

// Atributos e classes
elemento.setAttribute('data-info', 'valor');
elemento.classList.add('nova-classe');
elemento.classList.remove('classe-antiga');
elemento.classList.toggle('ativa');

// Estilos

const novoElemento = document.createElement('div');

elementoPai.removeChild(elementoFilho);

Eventos

Eventos são ações ou ocorrências que acontecem no navegador e podem ser detectadas pelo JavaScript.

Tipos Comuns de Eventos

  • Mouse: click, dblclick, mouseover, mouseout
  • Teclado: keydown, keyup, keypress
  • Formulário: submit, change, focus, blur
  • Janela: load, resize, scroll
  • Toque: touchstart, touchmove, touchend

Manipulação de Eventos


// Método tradicional
elemento.onclick = function() {
  console.log('Clique 1');
};

// addEventListener (recomendado)
elemento.addEventListener('click', function(event) {
  console.log('Clique 2', event.target);
});

// Remover listener
function handleClick() {
  console.log('Clique 3');
}
elemento.addEventListener('click', handleClick);
elemento.removeEventListener('click', handleClick);

// Prevenir comportamento padrão
formulario.addEventListener('submit', function(e) {
  e.preventDefault();
  // processar form manualmente
});

Propagação de Eventos


// Bubbling (padrão) - do elemento filho para os pais
// Capturing - dos pais para o elemento filho

document.querySelector('div').addEventListener('click', function() {
  console.log('Div clicada');
}, true); // true para capturing phase

document.querySelector('button').addEventListener('click', function(e) {
  console.log('Botão clicado');
  e.stopPropagation(); // impede a propagação
});

ES6+ Features

ECMAScript 2015 (ES6) trouxe muitas melhorias para a linguagem JavaScript.

Let e Const


// let - escopo de bloco
let x = 10;
if (true) {
  let x = 20;
  console.log(x); // 20
}
console.log(x); // 10

// const - constante (valor não pode ser reatribuído)
const PI = 3.1415;
PI = 3; // TypeError

Template Strings


const nome = 'João';
const idade = 30;

// String tradicional
console.log('Meu nome é ' + nome + ' e tenho ' + idade + ' anos.');

// Template string
console.log(`Meu nome é ${nome} e tenho ${idade} anos.`);

// Multilinha
const html = `
  <div>
    <h1>${nome}</h1>
    <p>Idade: ${idade}</p>
  </div>
`;

Arrow Functions


// Sintaxe tradicional
function soma(a, b) {
  return a + b;
}

// Arrow function
const soma = (a, b) => a + b;

// Sem parênteses para um parâmetro
const quadrado = x => x * x;

// Com corpo e return
const saudacao = nome => {
  const mensagem = `Olá, ${nome}!`;
  return mensagem;
};

// this léxico (não cria seu próprio this)
const relogio = {
  hora: 12,
  mostrarHora: function() {
    setInterval(() => {
      console.log(this.hora); // this refere-se a relogio
    }, 1000);
  }
};

Desestruturação


// Objetos
const pessoa = { nome: 'Maria', idade: 25 };
const { nome, idade } = pessoa;
console.log(nome); // Maria

// Arrays
const numeros = [1, 2, 3];
const [primeiro, segundo] = numeros;
console.log(segundo); // 2

// Parâmetros de função
function mostrarDados({ nome, idade }) {
  console.log(`${nome}, ${idade} anos`);
}
mostrarDados(pessoa); // Maria, 25 anos

// Valores padrão
const { cidade = 'Desconhecida' } = pessoa;
console.log(cidade); // Desconhecida

Async/Await

JavaScript é single-threaded, mas usa callbacks, promises e async/await para operações assíncronas.

Callbacks


function buscarDados(callback) {
  setTimeout(() => {
    callback('Dados recebidos');
  }, 1000);
}

buscarDados(function(dados) {
  console.log(dados); // Depois de 1 segundo
});

// Problema: Callback Hell
buscarUsuario(id, function(usuario) {
  buscarPosts(usuario, function(posts) {
    buscarComentarios(posts[0], function(comentarios) {
      console.log(comentarios);
    });
  });
});

Promises


function buscarDados() {
  return new Promise((resolve, reject) => {
    setTimeout(() => {
      resolve('Dados recebidos');
      // ou reject(new Error('Falha'));
    }, 1000);
  });
}

buscarDados()
  .then(dados => console.log(dados))
  .catch(erro => console.error(erro));

// Encadeamento
buscarUsuario(id)
  .then(usuario => buscarPosts(usuario))
  .then(posts => buscarComentarios(posts[0]))
  .then(comentarios => console.log(comentarios))
  .catch(erro => console.error(erro));

Async/Await (ES2017)


// Função assíncrona
async function processarDados() {
  try {
    const dados = await buscarDados();
    console.log(dados);
    
    const usuario = await buscarUsuario(dados.id);
    const posts = await buscarPosts(usuario.id);
    const comentarios = await buscarComentarios(posts[0].id);
    
    return comentarios;
  } catch (erro) {
    console.error('Erro:', erro);
  }
}

// IIFE (Immediately Invoked Function Expression)
(async () => {
  const resultado = await processarDados();
  console.log(resultado);
})();

Projeto Prático: To-Do List

Vamos criar uma aplicação simples de lista de tarefas para praticar os conceitos aprendidos.

Estrutura do Projeto


todo-list/
├── index.html
├── css/
│   └── estilo.css
└── js/
    └── app.js

Código do index.html


<!DOCTYPE html>
<html lang="pt-BR">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Lista de Tarefas</title>
  <link rel="stylesheet" href="css/estilo.css">
</head>
<body>
  <div class="container">
    <h1>Minha Lista de Tarefas</h1>
    
    <form id="formTarefa">
      <input type="text" id="inputTarefa" placeholder="Nova tarefa..." required>
      <button type="submit">Adicionar</button>
    </form>
    
    <ul id="listaTarefas"></ul>
  </div>
  
  <script src="js/app.js"></script>
</body>
</html>

Código do app.js


// Seleção de elementos
const form = document.getElementById('formTarefa');
const input = document.getElementById('inputTarefa');
const lista = document.getElementById('listaTarefas');

// Array para armazenar tarefas
let tarefas = JSON.parse(localStorage.getItem('tarefas')) || [];

// Função para salvar no localStorage
function salvarTarefas() {
  localStorage.setItem('tarefas', JSON.stringify(tarefas));
}

// Função para renderizar tarefas
function renderizarTarefas() {
  lista.innerHTML = '';
  
  tarefas.forEach((tarefa, index) => {
    const li = document.createElement('li');
    li.innerHTML = `
      <span class="${tarefa.concluida ? 'concluida' : ''}">${tarefa.texto}</span>
      <div class="acoes">
        <button onclick="toggleConcluida(${index})">
          ${tarefa.concluida ? 'Desfazer' : 'Concluir'}
        </button>
        <button onclick="removerTarefa(${index})">Remover</button>
      </div>
    `;
    lista.appendChild(li);
  });
}

// Adicionar nova tarefa
form.addEventListener('submit', (e) => {
  e.preventDefault();
  
  const texto = input.value.trim();
  if (texto) {
    tarefas.push({ texto, concluida: false });
    salvarTarefas();
    renderizarTarefas();
    input.value = '';
    input.focus();
  }
});

// Alternar status de conclusão
function toggleConcluida(index) {
  tarefas[index].concluida = !tarefas[index].concluida;
  salvarTarefas();
  renderizarTarefas();
}

// Remover tarefa
function removerTarefa(index) {
  tarefas.splice(index, 1);
  salvarTarefas();
  renderizarTarefas();
}

// Inicializar
renderizarTarefas();
